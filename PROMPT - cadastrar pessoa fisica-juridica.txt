-- pessoas_fisicas (UNIQUE KEY uq_pf_cpf em cpf)
pessoas_fisicas (
  id, nome, cpf, rg, rg_orgao, pis, ctps_numero, ctps_serie,
  nome_pai, nome_mae, data_nascimento,
  estado_civil_id, profissao_id, genero_id, nacionalidade_id,
  cep, logradouro, numero, complemento, bairro, cidade, estado,
  foto_path, observacoes, ativo, criado_em, criado_por, alterado_por, alterado_em
)

-- pessoas_juridicas (UNIQUE KEY uq_pj_cnpj em cnpj)
pessoas_juridicas (
  id, razao_social, nome_fantasia, cnpj, inscricao_estadual,
  cep, logradouro, numero, complemento, bairro, cidade, estado,
  observacoes, ativo, criado_em, criado_por, alterado_por, alterado_em
)

-- telefones_pf (FK pessoa_id -> pessoas_fisicas.id)
telefones_pf (
  id, pessoa_id, numero, tipo (default 'Celular'), principal, ativo, criado_em
)

-- telefones_pj (FK pessoa_id -> pessoas_juridicas.id)
telefones_pj (
  id, pessoa_id, numero, tipo (default 'Comercial'), principal, ativo, criado_em
)

-- emails_pf (FK pessoa_id -> pessoas_fisicas.id)
emails_pf (
  id, pessoa_id, email, principal, ativo, criado_em
)

-- emails_pj (FK pessoa_id -> pessoas_juridicas.id)
emails_pj (
  id, pessoa_id, email, principal, ativo, criado_em
)

-- tabelas auxiliares (todas: id, nome)
genero (id, nome)
estado_civil (id, nome)
profissao (id, nome)
nacionalidade (id, nome)   -- "Brasileira" já existe
